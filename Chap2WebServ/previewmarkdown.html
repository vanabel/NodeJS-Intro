第六节. 预览与Markdown排版

我们已经看到，编辑器可以输入内容，但是别忘了我们的目标是要在右端预览输入的内容。我们首先来看看客户端的实现方式，即如何在浏览器中利用Javascript脚本实现实时预览。

使用Javascript脚本实时预览

我们其实可以监听用户输入事件(key)，使得事件触发时我们执行更新用户输入到预览。这用纯JavaScript也是非常容易实现的。下面的代码，我们还设置了一个延迟函数（防止更新到预览这个过程有好几个进程同时执行）。代码中省略了原来index.html文件中相同部分。

<script>
function set(el, text){
    while (el.firstChild) {
        el.removeChild(el.firstChild);
    }
    el.appendChild(document.createTextNode(text));
}

function update(){
    var text = document.getElementById('editor');
    var oldText = text.value;
    var preview = document.getElementById('preview');
    var timeout = null;

    function handleChange(){
        var newText=text.value;
        if (newText==oldText) return; else oldText=newText;
        set(preview, newText);
    }

    function handle(){
        if(timeout) clearTimeout(timeout);
        timeout=setTimeout(handleChange, 50);
    }

    text.onkeydown=text.onkeyup=text.onclick=handle;
}

update();
document.getElementById('editor').focus();
</script>

使用Markdown排版

别忘了，我们的目标是实现一个使用markdown语法的编辑器。上面的代码只能照原来的内容输出到预览。下面我们将通过showdownjs这个插件实现markdown的实时预览。我们首先还是来看以前怎么做的？即将showdownjs作为文件引入到我们的index.html， 然后在自定义的usr.js中使用该插件实现makrdown排版。

首先，我们用npm安装showdownjs.

npm install showdownjs

这会在src/目录下创建node_modules目录，Node的模块都将安装到该目录。我们新的 index.html 将从这个目录包含进来 showdown.js. 这个过程是很简单的（我省略了index.html中的相同代码）, 即在</body>标签之前加入如下javascript脚本即可。

<script src='node_modules/showdown/dist/showdown.min.js'></script>
<script src='usr.js'></script>

这里顺序是重要的，而usr.js其实就是我们将原来嵌入到index.html的脚本改写后用showdown.js实现markdown排版的脚本文件。关于showdown.js的用法可以参考Tutorial: Markdown editor using showdown.

// usr.js

function update(){
    var text = document.getElementById('editor');
    var oldText = text.value;
    var preview = document.getElementById('preview');
    var converter = new showdown.Converter();
    var timeout = null;

    function handleChange(){
        var newText=text.value;
        if (newText === oldText) {
            return;
        } else {
            oldText=newText;
        }
        var html = converter.makeHtml(newText);
        preview.innerHTML = html;
    }

    function handle(){
        if (timeout) {
            clearTimeout(timeout);
        }
        timeout=setTimeout(handleChange, 50);
    }

    text.onkeydown=text.onkeyup=text.onclick=handle;
}

update();
document.getElementById('editor').focus();

好了，你是不是和我一样感觉应该行了？运行会发现（浏览器控制台）我们在index.html中用到的两个脚本找不到。这时当然的，因为我们没有给这两个文件路由啊！下面我们来改写路由处理函数，使得它不仅可以读取index.html, 而且还可以读取其他文件， 例如javascript脚本与css样式表文件。

// file: routeHander
var fs = require('fs');
var path = require('path');
var html = 'The html content.';

function get(fpath, res){
    console.log('Get handler is called.');

    if (fpath === '/' ) {
        fpath = '/index.html';
    }
    if (path.extname(fpath) === '.js') {
        res.writeHead(200, {'content-type': 'text/javascript'});
    }
    if (path.extname(fpath) === '.css') {
        res.writeHead(200, {'content-type': 'text/css'});
    }
    if (path.extname(fpath) === '.html') {
        res.writeHead(200, {'content-type': 'text/html; charset=utf-8'});
    }

    fs.readFile(__dirname + fpath, function(err, data){
        if (err)  {
            throw err;
        }
        html = data.toString();
        console.log(fpath);
        res.write(html);
        res.end();
    });
}

function post(fpath, res){
    console.log('Post handler is called.');
    res.writeHead(200, {'content-type': 'text/html; charset=utf-8'});
    res.write('Hello, world!');
    res.end();
}

exports.get = get;
exports.post = post;

这里关键是处理了头信息, 并将我们的/路径映射到真实文件/index.html. 由于我们改变了路由处理函数， 特别地它将接受新的fpath参数，即文件路径，那么我们也要相应的更改路由.


// file: router.js

var path = require('path');

function route(handle, fpath, res){
    var hpath = fpath,
        ext = path.extname(hpath),
        extarr = ['.js', '.map', '.css'];
    if ( extarr.indexOf(ext)>-1 ) {
        hpath = '/getfile';
    }
    console.log('Handle request from: ' + fpath);
    if (typeof handle[hpath] === 'function' ){
        return handle[hpath](fpath, res);
    } else {
        console.log('404 not found');
        res.writeHead(404, {'content-type': '404 not found'});
        res.end();
    }
}

exports.route = route;

这里， 我们定义了一个新的（伪）路由：/getfile, 它其实就是就是告诉路由处理函数，以.js, .map, .css为后缀的文件需要用/getfile这个路由处理函数来处理。具体地，这个路由的映照是如下定义的：

// file: index.js
var server=require('./server');
var router=require('./router');
var handler=require('./routeHandler');

var handle = {};
handle['/'] = handler.get;
handle['/getfile'] = handler.get;
handle['/post'] = handler.post;

server.init(handle, router.route);

如此，我们完成了服务器处理文件这一改造。重新运行， 你应该能过用Markdown来编辑了。

最后， 作为练习， 请将我们原来包含在index.html中的样式表独立出来放到style.css，看看我们上述改造后的服务器能不能正常处理style.css.
